CREATE TABLE clientes (
    id NUMBER PRIMARY KEY NOT NULL,
    nome VARCHAR2(50) NOT NULL
);

    v_id NUMBER:= 1;
    v_novo_nome VARCHAR2(50):='Maria da Silva';
BEGIN
    UPDATE clientes set nome = v_novo_nome WHERE id = v_id;
    COMMIT;        
END;
