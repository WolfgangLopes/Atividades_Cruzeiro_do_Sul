CREATE TABLE alunos (

  id_alunos number,

  nome varchar(50),

  CONSTRAINT alunos_id_alunos_pk PRIMARY KEY (id_alunos)

);


CREATE TABLE matricula(

  id_matricula number,

  id_alunos number,

  curso varchar(50),

  CONSTRAINT matricula_id_matricula_pk PRIMARY KEY (id_matricula),

  CONSTRAINT matricula_id_aluno_fk FOREIGN KEY (id_alunos) REFERENCES aluno(id_alunos)

  ON DELETE CASCADE

  );
